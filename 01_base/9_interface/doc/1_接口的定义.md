# 1.接口的定义
```properties
// 1.接口的声明
type interface_name interface {
    // 声明的方法
    methodName1(paramName1 paramType, ...) (returnVal1 returnType, ...)
    ...
    methodNameN(paramNameN paramType, ...) (returnValN returnType, ...)
}

// 2.定义实现接口的结构体
type struct_name struct {
    ...
}

// 3.绑定接口和结构体（也叫结构体实现接口），即实现接口声明的所有方法 + 结构体的指针接收器
func (sn *struct_name) methodName1(paramName1 paramType, ...) (returnVal1 returnType, ...){
    ...
    return ...
}

...

func (sn *struct_name) methodNameN(paramNameN paramType, ...) (returnValN returnType, ...){
    ...
    return ...
}

4.使用，需要使用结构体取值
var interface_var interface_name = &struct_name{}
```
